CREATE TABLE achados_perdidos
(
  id_item  		SERIAL,
  id_pessoa		SERIAL,
  nm_item          	VARCHAR(40),
  local_perdido   VARCHAR(20),
  data_perdido    	DATE,
  observacao VARCHAR(40),
  Status_Achado     boolean,	
  PRIMARY KEY(id_item)
);
